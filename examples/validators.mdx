Script addresses are addressed by a `policy` declaration. A script input carries a `redeemer`; a `datum_is` annotation lets the rest of the transaction read the datum as typed fields. Plutus inputs also require a `collateral` block.

## Vesting: lock then unlock

Two transactions sharing the same `TimeLock` policy. `lock` produces an output at the script address carrying a `State` datum; `unlock` consumes it back.

```tx3
party Owner;

party Beneficiary;

policy TimeLock = 0x6b9c456aa650cb808a9ab54326e039d5235ed69f069c9664a8fe5b69;

type State {
  lock_until: Int,
  owner: Bytes,
  beneficiary: Bytes,
}

tx lock(
    quantity: Int,
    until: Int
) {
    input source {
        from: Owner,
        min_amount: Ada(quantity),
    }

    output target {
        to: TimeLock,
        amount: Ada(quantity),
        datum: State {
            lock_until: until,
            owner: Owner,
            beneficiary: Beneficiary,
        },
    }

    output {
        to: Owner,
        amount: source - Ada(quantity) - fees,
    }
}

tx unlock(
    locked_utxo: UtxoRef
) {
    input gas {
        from: Beneficiary,
        min_amount: fees,
    }

    input locked {
        from: TimeLock,
        ref: locked_utxo,
        redeemer: (),
    }

    collateral {
        from: Beneficiary,
        min_amount: fees,
    }

    output target {
        to: Beneficiary,
        amount: gas + locked - fees,
    }
}
```

`unlock` takes the locked UTxO as a `UtxoRef` parameter — the caller passes the on-chain reference of whatever output `lock` produced. `redeemer: ()` is the unit value, used when the validator does not need data from the spender.

## Reading and updating an input datum

When `datum_is` is set on an input, you can read its fields via property access (`source.counter`). Combined with the spread operator `...`, this is the standard pattern for updating on-chain state: copy the existing record, overriding only the fields that change.

```tx3
type MyRecord {
    counter: Int,
    other_field: Bytes,
}

party MyParty;

tx increase_counter() {
    input source {
        from: MyParty,
        min_amount: fees,
        datum_is: MyRecord,
    }

    output {
        to: MyParty,
        amount: source - fees,
        datum: MyRecord {
            counter: source.counter + 1,
            other_field: source.other_field,
        },
    }
}
```

## A pool swap

A DEX swap shows how the input's typed datum threads through both the redeemer and a spread-merged output datum. `pool.pair_a` and `pool.pair_b` access fields of the validator's `PoolState`; the new output reuses unchanged fields with `...pool`.

```tx3
type PoolState {
    pair_a: AnyAsset,
    pair_b: AnyAsset,
}

type SwapParams {
    ask_value: AnyAsset,
    bid_value: AnyAsset,
}

party Buyer;

party Dex;

tx swap(
    ask: AnyAsset,
    bid: AnyAsset,
) {
    input pool {
        from: Dex,
        datum_is: PoolState,

        redeemer: SwapParams {
            ask_value: ask,
            bid_value: bid,
        },
    }

    input payment {
        from: Buyer,
        min_amount: fees + bid,
    }

    output {
        to: Dex,
        datum: PoolState {
            pair_a: pool.pair_a - bid.amount,
            pair_b: pool.pair_b + ask.amount,
            ...pool
        },
        amount: pool,
    }

    output {
        to: Buyer,
        amount: payment + ask - bid - fees,
    }
}
```
